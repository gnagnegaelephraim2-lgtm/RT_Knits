## Phase 3: JavaScript Rewrite (`public/app.js`)

### 3.1 Rewrite `initAuthGate()` — Server-Side Auth

Replace the login submit handler to call the Rust `/api/auth/login` endpoint:

```javascript
function initAuthGate() {
  const savedSession = localStorage.getItem("nita_active_session");
  if (savedSession) {
    activeSession = JSON.parse(savedSession);
    applyRolePermissions(activeSession);
  } else {
    showAuthOverlay(true);
  }

  // Preset button selection
  document.querySelectorAll('.preset-btn').forEach(btn => {
    btn.addEventListener('click', () => {
      document.querySelectorAll('.preset-btn').forEach(b => b.classList.remove('selected'));
      btn.classList.add('selected');
      document.getElementById('auth-phone').value = btn.dataset.phone;
      document.getElementById('auth-pin').value = btn.dataset.pin;
      // Auto-submit after brief delay for visual feedback
      setTimeout(() => handleLogin(), 300);
    });
  });

  // Form submit
  document.getElementById('auth-form')?.addEventListener('submit', (e) => {
    e.preventDefault();
    handleLogin();
  });

  // Logout
  document.getElementById('btn-logout')?.addEventListener('click', () => {
    activeSession = null;
    localStorage.removeItem("nita_active_session");
    showAuthOverlay(true);
  });
}

async function handleLogin() {
  const phone = document.getElementById('auth-phone').value.trim();
  const pin = document.getElementById('auth-pin').value.trim();
  const card = document.getElementById('auth-card');
  const btn = document.getElementById('btn-login-submit');

  if (!phone || !pin) {
    card?.classList.add('shake');
    setTimeout(() => card?.classList.remove('shake'), 500);
    return;
  }

  // Hash PIN with WebCrypto SHA-256
  const pinHash = await sha256Hex(pin);

  // Show loading state
  btn?.classList.add('loading');
  btn.disabled = true;

  try {
    const res = await fetch('/api/auth/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ phone_number: phone, pin_hash: pinHash })
    });
    const data = await res.json();

    if (data.error) {
      card?.classList.add('shake');
      setTimeout(() => card?.classList.remove('shake'), 500);
      return;
    }

    // Store JWT and session
    localStorage.setItem('nita_jwt_token', data.token);
    const selectedBtn = document.querySelector('.preset-btn.selected');
    activeSession = {
      name: selectedBtn?.dataset.name || 'User',
      phone: phone,
      role: data.role
    };
    localStorage.setItem("nita_active_session", JSON.stringify(activeSession));
    applyRolePermissions(activeSession);
    showAuthOverlay(false);
  } catch (err) {
    card?.classList.add('shake');
    setTimeout(() => card?.classList.remove('shake'), 500);
  } finally {
    btn?.classList.remove('loading');
    btn.disabled = false;
  }
}

// WebCrypto SHA-256 helper
async function sha256Hex(message) {
  const msgBuffer = new TextEncoder().encode(message);
  const hashBuffer = await crypto.subtle.digest('SHA-256', msgBuffer);
  const hashArray = Array.from(new Uint8Array(hashBuffer));
  return hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
}
```

### 3.2 Rewrite `showAuthOverlay()` — Animated Transitions

```javascript
function showAuthOverlay(show) {
  const overlay = document.getElementById('auth-overlay');
  if (!overlay) return;

  if (show) {
    overlay.classList.remove('hidden', 'animating-out');
    overlay.style.display = 'flex';
  } else {
    overlay.classList.add('animating-out');
    setTimeout(() => {
      overlay.classList.add('hidden');
      overlay.classList.remove('animating-out');
    }, 350);
  }
}
```
